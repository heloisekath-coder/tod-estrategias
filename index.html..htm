  <!-- Importar Three.js e OrbitControls como módulos (CDN) -->

  <script type="module">

    // Importes via CDN (module build)

    import * as THREE from 'https://unpkg.com/three@0.154.0/build/three.module.js';

    import { OrbitControls } from 'https://unpkg.com/three@0.154.0/examples/jsm/controls/OrbitControls.js';

 

    // ----- Setup básico -----

    const canvas = document.getElementById('canvas3d');

    const renderer = new THREE.WebGLRenderer({ canvas, antialias: true, alpha: true });

    renderer.setPixelRatio(window.devicePixelRatio ? Math.min(window.devicePixelRatio, 2) : 1);

    renderer.setSize(canvas.clientWidth, canvas.clientHeight, false);

    renderer.outputEncoding = THREE.sRGBEncoding;

 

    const scene = new THREE.Scene();

 

    // Câmera: foque em uma cena de slide (campo de visão confortável)

    const camera = new THREE.PerspectiveCamera(45, canvas.clientWidth / canvas.clientHeight, 0.1, 100);

    camera.position.set(0, 1.6, 5);

 

    // Controles de órbita (interatividade com mouse)

    const controls = new OrbitControls(camera, renderer.domElement);

    controls.enableDamping = true;

    controls.dampingFactor = 0.08;

    controls.minDistance = 2;

    controls.maxDistance = 12;

    controls.target.set(0, 1, 0.1);

 

    // ----- Iluminação -----

    const ambient = new THREE.HemisphereLight(0xffffff, 0xbfd8ff, 0.8); // luz ambiente suave

    scene.add(ambient);

 

    const dir = new THREE.DirectionalLight(0xffffff, 0.8);

    dir.position.set(5, 10, 7);

    dir.castShadow = false;

    scene.add(dir);

 

    // Sutil brilho frontal para destacar material

    const keyFill = new THREE.PointLight(0xffffff, 0.15);

    keyFill.position.set(-3, 3, -3);

    scene.add(keyFill);

 

    // ----- Helpers de responsividade -----

    function resizeRendererToDisplaySize() {

      const width = canvas.clientWidth;

      const height = canvas.clientHeight;

      const needResize = canvas.width !== Math.floor(width * renderer.getPixelRatio()) || canvas.height !== Math.floor(height * renderer.getPixelRatio());

      if (needResize) {

        renderer.setSize(width, height, false);

        camera.aspect = width / height;

        camera.updateProjectionMatrix();

      }

      return needResize;

    }

 

    // ----- Materiais leves e acessíveis (contraste bom) -----

    const materialFamily = new THREE.MeshStandardMaterial({ color: 0x2c6ea4, roughness: 0.6, metalness: 0.05 });

    const materialFamilyAccent = new THREE.MeshStandardMaterial({ color: 0x76b5dd, roughness: 0.6 });

    const materialBrain = new THREE.MeshStandardMaterial({ color: 0xffd59a, roughness: 0.7 });

    const materialLobo = new THREE.MeshStandardMaterial({ color: 0xffb86b, roughness: 0.6 });

    const materialArrow = new THREE.MeshStandardMaterial({ color: 0xffffff, roughness: 0.4 });

 

    // ----- Criação da "família" central com geometrias simples -----

    const familyGroup = new THREE.Group();

    familyGroup.position.set(-0.8, 0.6, 0);

 

    // Função auxiliar que cria uma "pessoa" simples: cilindro corpo + esfera cabeça

    function createPerson(scale = 1, bodyColor = materialFamily, headColor = materialFamilyAccent) {

      const g = new THREE.Group();

 

      const bodyGeom = new THREE.CylinderGeometry(0.22 * scale, 0.28 * scale, 0.8 * scale, 12);

      const body = new THREE.Mesh(bodyGeom, bodyColor);

      body.position.y = 0.3 * scale;

      g.add(body);

 

      const headGeom = new THREE.SphereGeometry(0.2 * scale, 12, 10);

      const head = new THREE.Mesh(headGeom, headColor);

      head.position.y = 0.95 * scale;

      g.add(head);

 

      // braços simples (torus ou cilindros)

      const armGeom = new THREE.CylinderGeometry(0.05 * scale, 0.05 * scale, 0.6 * scale, 8);

      const leftArm = new THREE.Mesh(armGeom, bodyColor);

      leftArm.rotation.z = Math.PI / 6;

      leftArm.position.set(-0.3 * scale, 0.5 * scale, 0);

      g.add(leftArm);

 

      const rightArm = leftArm.clone();

      rightArm.rotation.z = -Math.PI / 6;

      rightArm.position.x = 0.3 * scale;

      g.add(rightArm);

 

      return g;

    }

 

    // Pai, mãe e criança em círculo (poses positivas)

    const pai = createPerson(1.0, materialFamily, materialFamilyAccent);

    pai.position.set(-0.7, 0, 0);

    pai.rotation.y = 0.2;

    familyGroup.add(pai);

 

    const mae = createPerson(1.0, materialFamilyAccent, materialFamily);

    mae.position.set(0.0, 0, -0.45);

    mae.rotation.y = 1.2;

    familyGroup.add(mae);

 

    const crianca = createPerson(0.7, materialFamily, materialFamilyAccent);

    crianca.position.set(0.6, 0, 0);

    crianca.rotation.y = -0.6;

    familyGroup.add(crianca);

 

    // Pequeno tapete (círculo) para indicar encontro/terapia familiar

    const rug = new THREE.Mesh(new THREE.CircleGeometry(1.2, 48), new THREE.MeshStandardMaterial({ color: 0xdff7fb, roughness: 1 }));

    rug.rotation.x = -Math.PI / 2;

    rug.position.set(0, 0.01, 0); // leve acima do chão

    scene.add(rug);

 

    scene.add(familyGroup);

 

    // ----- "Piscar" suave da família (animação de material emissive) -----

    const familyBlink = {

      elapsed: 0,

      speed: 1.2 // segundos por ciclo

    };

 

    // ----- Cérebro estilizado à direita (esfera com "lobos" destacados) -----

    const brainGroup = new THREE.Group();

    brainGroup.position.set(1.6, 1.2, 0.2);

 

    // cérebro base (esfera)

    const brainBase = new THREE.Mesh(new THREE.SphereGeometry(0.7, 32, 24), materialBrain);

    brainGroup.add(brainBase);

 

    // destacar lobos: adiciona "protuberâncias" via esferas menores

    const loboOffsets = [

      [0.45, 0.15, 0.4],

      [-0.1, 0.28, 0.6],

      [-0.45, -0.05, 0.35],

      [0.2, -0.28, 0.45],

    ];

    loboOffsets.forEach((off, i) => {

      const s = new THREE.Mesh(new THREE.SphereGeometry(0.18 + i * 0.03, 16, 12), materialLobo);

      s.position.set(off[0], off[1], off[2]);

      brainGroup.add(s);

    });

 

    // leve rotação contínua para simbolizar regulação emocional

    scene.add(brainGroup);

 

    // ----- Setas / cubos flutuantes representando estratégias -----

    const strategies = [

      { label: 'Treinamento Parental', pos: [0.5, 2.0, -1.2], color: 0x1f77b4 },

      { label: 'Resolução de Problemas', pos: [-2.3, 1.8, -0.3], color: 0x2ca02c },

      { label: 'Limites Claros', pos: [-0.6, 2.1, 1.4], color: 0xd62728 }

    ];

 

    const strategyGroup = new THREE.Group();

    scene.add(strategyGroup);

 

    // Função para criar um "bloco/ícone" com texto como textura

    function createStrategyNode(text, colorHex) {

      const node = new THREE.Group();

 

      // cubo/selo

      const box = new THREE.Mesh(new THREE.BoxGeometry(0.6, 0.3, 0.2), new THREE.MeshStandardMaterial({ color: colorHex, roughness: 0.5 }));

      box.castShadow = true;

      node.add(box);

 

      // seta 3D simples (pirâmide) apontando para a família

      const coneGeom = new THREE.ConeGeometry(0.12, 0.28, 8);

      const cone = new THREE.Mesh(coneGeom, materialArrow);

      cone.rotation.x = Math.PI;

      cone.position.set(0, -0.28, 0);

      node.add(cone);

 

      // texto como textura (canvas)

      const canvasTxt = document.createElement('canvas');

      const ctx = canvasTxt.getContext('2d');

      canvasTxt.width = 512;

      canvasTxt.height = 128;

      // fundo semi-transparente para garantir legibilidade

      ctx.fillStyle = 'rgba(0,0,0,0)';

      ctx.fillRect(0, 0, canvasTxt.width, canvasTxt.height);

      ctx.fillStyle = '#ffffff';

      ctx.textAlign = 'center';

      ctx.font = '28px sans-serif';

      wrapText(ctx, text, canvasTxt.width / 2, 40, 480, 30);

 

      const tex = new THREE.CanvasTexture(canvasTxt);

      tex.encoding = THREE.sRGBEncoding;

 

      const plane = new THREE.Mesh(new THREE.PlaneGeometry(1.0, 0.26), new THREE.MeshBasicMaterial({ map: tex, transparent: true }));

      plane.position.set(0, 0.4, 0);

      node.add(plane);

 

      return node;

    }

 

    // função auxiliar para quebra de linha no canvas

    function wrapText(ctx, text, x, y, maxWidth, lineHeight) {

      const words = text.split(' ');

      let line = '';

      for (let n = 0; n < words.length; n++) {

        const testLine = line + words[n] + ' ';

        const metrics = ctx.measureText(testLine);

        const testWidth = metrics.width;

        if (testWidth > maxWidth && n > 0) {

          ctx.fillText(line, x, y);

          line = words[n] + ' ';

          y += lineHeight;

        } else {

          line = testLine;

        }

      }

      ctx.fillText(line, x, y);

    }

 

    // Criar e posicionar nodes

    strategies.forEach((s, i) => {

      const node = createStrategyNode(s.label, s.color);

      node.position.set(s.pos[0], s.pos[1], s.pos[2]);

      // guardar velocidade/orbita para animação (cada um com fase diferente)

      node.userData = { orbitRadius: 0.22 + i * 0.05, orbitSpeed: 0.6 + i * 0.2, phase: i * Math.PI / 3 };

      strategyGroup.add(node);

    });

 

    // ----- Título 3D flutuante no topo (texto como textura em plano) -----

    function createFloatingTitle(mainText, subtitle) {

      const group = new THREE.Group();

 

      // Canvas para o título

      const cTitle = document.createElement('canvas');

      cTitle.width = 1600;

      cTitle.height = 240;

      const ct = cTitle.getContext('2d');

 

      // Fundo transparente -> texto branco grande

      ct.clearRect(0, 0, cTitle.width, cTitle.height);

      ct.font = '48px "Helvetica Neue", Arial, sans-serif';

      ct.fillStyle = '#ffffff';

      ct.textAlign = 'center';

      ct.fillText(mainText, cTitle.width / 2, 80);

 

      ct.font = '20px "Helvetica Neue", Arial, sans-serif';

      ct.fillStyle = '#f5fbff';

      ct.fillText(subtitle, cTitle.width / 2, 140);

 

      const tex = new THREE.CanvasTexture(cTitle);

      tex.encoding = THREE.sRGBEncoding;

      const pl = new THREE.Mesh(new THREE.PlaneGeometry(6.0, 0.95), new THREE.MeshBasicMaterial({ map: tex, transparent: true }));

      pl.position.set(0, 3.0, -0.2);

      group.add(pl);

 

      return group;

    }

 

    const titleGroup = createFloatingTitle(

      "Estratégias para TOD (DSM-5-TR)",

      "Reduza irritabilidade e conflitos com intervenções baseadas em evidências"

    );

    scene.add(titleGroup);

 

    // ----- Rodapé / referências rápidas (plano 3D) -----

    (function createFooter() {

      const c = document.createElement('canvas');

      c.width = 1024;

      c.height = 80;

      const ctx = c.getContext('2d');

      ctx.clearRect(0, 0, c.width, c.height);

      ctx.font = '18px Arial';

      ctx.fillStyle = '#013049'; // contraste com fundo

      ctx.textAlign = 'center';

      ctx.fillText('Baseado em APA (2022) e AACAP (2019)', c.width / 2, 48);

 

      const t = new THREE.CanvasTexture(c);

      t.encoding = THREE.sRGBEncoding;

      const mesh = new THREE.Mesh(new THREE.PlaneGeometry(4.6, 0.36), new THREE.MeshBasicMaterial({ map: t, transparent: true }));

      mesh.position.set(0, -0.6, 0.8);

      scene.add(mesh);

    })();

 

    // ----- Animação: orbitar setas, girar cérebro, piscar família -----

    const clock = new THREE.Clock();

 

    function animate() {

      requestAnimationFrame(animate);

 

      const dt = clock.getDelta();

      const t = clock.getElapsedTime();

 

      // Responsividade

      resizeRendererToDisplaySize();

 

      // Atualizar controles

      controls.update();

 

      // Animação do cérebro (rotação lenta)

      brainGroup.rotation.y += dt * 0.2;

 

      // Animação das nodes (orbitar levemente em torno de seus pontos)

      strategyGroup.children.forEach((node, idx) => {

        const ud = node.userData;

        const angle = t * ud.orbitSpeed + ud.phase;

        node.position.x += Math.sin(angle) * 0.002; // pequenas oscilações

        node.position.y += Math.cos(angle * 1.1) * 0.002;

        node.rotation.y += dt * 0.2; // rotação lenta do próprio node para dinamismo

      });

 

      // Família "pisca": animação de emissive ou escala muito sutil

      familyBlink.elapsed += dt;

      const blinkProgress = (Math.sin((familyBlink.elapsed / familyBlink.speed) * Math.PI * 2) + 1) / 2; // 0..1

      // aplicamos uma pequena mudança no emissive/intensidade (simples)

      const emissiveIntensity = 0.04 * blinkProgress;

      [pai, mae, crianca].forEach(person => {

        person.traverse((m) => {

          if (m.isMesh && m.material && m.material.emissive !== undefined) {

            m.material.emissive = new THREE.Color(0x66c2ff).multiplyScalar(emissiveIntensity);

          }

        });

        // leve "respiração" de escala

        person.scale.y = 1 + 0.005 * blinkProgress;

      });

 

      // render

      renderer.render(scene, camera);

    }

    animate();

 

    // ----- Acessibilidade / controles de teclado mínimos -----

    window.addEventListener('keydown', (e) => {

      // teclas +/- para zoom (acessibilidade)

      if (e.key === '+' || e.key === '=') {

        camera.position.z = Math.max(1.5, camera.position.z - 0.3);

      } else if (e.key === '-') {

        camera.position.z = Math.min(14, camera.position.z + 0.3);

      }

    });

 

    // Atualiza o tamanho do renderer quando a janela muda

    window.addEventListener('resize', () => {

      resizeRendererToDisplaySize();

    });

 

    // Prevenção de seleção e arrasto ao usar o slide em iframe/embeds

    renderer.domElement.addEventListener('pointerdown', (e) => {

      e.target.focus && e.target.focus();

    });

 

    // ---- Nota sobre performance ----

    // Mantivemos geometrias simples, materiais sem sombras pesadas e texturas de canvas (pequenas) para rodar em navegadores comuns.

    // Se quiser reduzir ainda mais o uso de GPU, diminua renderer.setPixelRatio(...) ou reduza segmentos nas esferas.

  </script>
